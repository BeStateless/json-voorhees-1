00307         if (!context.next())

00309             break;
00310         
00311         if (!allowed_chars.count(context.current))
00312         {
00313             context.previous();
00314             break;
00315         }
00316         else
00317         {
00318             if (context.current == '.')
00319                 is_double = true;
00320             
00321             ++characters_count;
00322         }
00323     }
00324     
00325     try
00326     {
00327         
00328         if (is_double)
00329             return boost::lexical_cast<double>(characters_start, characters_count);

00331             return boost::lexical_cast<int64_t>(characters_start, characters_count);

00336             return static_cast<int64_t>(boost::lexical_cast<uint64_t>(characters_start, characters_count));
